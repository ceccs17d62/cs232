float computeGPA(char buf[]) {
	int i, j, count=0;
	float total=0;
	char grade[3];
	for (i=0; buf[i] != '\0'; i++) {
		if (buf[i] == '(') {
			j=0;
			i++;
			count++;
			while(buf[i] != ')') {
				grade[j++] = buf[i++];
			}
			grade[j] = '\0';
			if (!strcmp(grade, "O")) {
				total += 10;
			} else if (!strcmp(grade, "A+")) {
				total += 9;
			} else if (!strcmp(grade, "A")) {
				total += 8.5;
			} else if (!strcmp(grade, "B+")) {
				total += 8;
			} else if (!strcmp(grade, "B")) {
				total += 7;
			} else if (!strcmp(grade, "C")) {
				total += 6;
			} else if (!strcmp(grade, "P")) {
				total += 5;
			} else {
				total += 0;
			}
		}
	}
	return (float)(total/count);
}
